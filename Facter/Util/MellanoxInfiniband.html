def self.count_ib_devices
  count = 0
  if Facter::Util::Resolution.which('lspci')
    output = lspci
    matches = output.scan(LSPCI_IB_REGEX)
    count = matches.flatten.reject { |s| s.nil? }.length
  end
  count
end
